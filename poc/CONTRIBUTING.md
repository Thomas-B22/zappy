# How to add a module to libs/

## Makefile

With any of the following languages, you must first add your module name in the `Makefile` as follow:
```make
<language>_MODULES =  \
  <module_name>
```

Here's how to compile the module only:
```bash
make <module_name>
```

For every language except Rust you need a `libs/<module_name>/.world` file containing only the module world:
```txt
<module_world>
```

## Rust

First make the Cargo crate:
```bash
cargo new --lib libs/<module_name>
```

Then edit `libs/<module_name>/Cargo.toml` file to look like this:
```toml
[package]
name = "<module_name>"
version = "0.1.0"
edition = "2024"

[lib]
crate-type = ["cdylib"]

[dependencies]
wit-bindgen = "0.57.1" # MANDATORY
serde = { version = "1.0.228", features = ["derive"] } # Optionnal: only if wit world needs serialize / deserialize
bincode = "1.3" # Optionnal: only if wit world needs serialize / deserialize

[package.metadata.component]
package = "local:zappy"

[package.metadata.component.target]
path = "../../wit"
world = "<module_world>"
```

Finally write your code inside `libs/<module_name>/src/lib.rs` with `wit-bindgen`:
```rust
wit_bindgen::generate!({
    path: "../../wit",
    world: "<module_world>"
});

struct Module;

impl Guest for Module {
  /* World functions here */
}
```


## Zig

Since Makefile does everything for Zig, it only needs a `libs/<module_name>/main.zig` file. e.g:
```zig
const std = @import("std");
const allocator = std.heap.wasm_allocator;

const c = @cImport({
    @cInclude("<module_world>.h");
});

// stdlib.h functions:

export fn malloc(size: usize) callconv(.c) ?*anyopaque {
    const slice = allocator.alloc(u8, size) catch return null;
    return slice.ptr;
}

export fn free(ptr: ?*anyopaque) callconv(.c) void {
    _ = ptr;
}

export fn realloc(ptr: ?*anyopaque, size: usize) callconv(.c) ?*anyopaque {
    if (size == 0) return null;
    if (ptr) |p| {
        const new_ptr = malloc(size);
        if (new_ptr) |n_ptr| {
            @memcpy(@as([*]u8, @ptrCast(n_ptr))[0..size], @as([*]u8, @ptrCast(p))[0..size]);
            return n_ptr;
        }
        return null;
    }
    return malloc(size);
}

export fn abort() callconv(.c) noreturn {
    while (true) {}
}

// WIT function example: exports_<module_world>_<function_name>(<parameter>: <type>) callconv(.c) <return type> {}
export fn exports_ui_world_handle_input(state_ptr: usize) callconv(.c) void {
    _ = state_ptr;
}
```

## C

Same as for Zig, it needs a `libs/<module_name>/main.c` file. e.g:
```c
#include "<module_world>.h"

#define HEAP_SIZE 65536
static uint8_t g_heap[HEAP_SIZE];
static unsigned long g_heap_idx = 0;

// stdlib.h functions:

void *memcpy(void *dest, const void *src, unsigned long n) {
    uint8_t *d = (uint8_t *)dest;
    const uint8_t *s = (const uint8_t *)src;
    for (unsigned long i = 0; i < n; i++) {
        d[i] = s[i];
    }
    return dest;
}

void *malloc(unsigned long size) {
    size = (size + 7) & ~7;
    if (g_heap_idx + size > HEAP_SIZE) {
        return (void*)0;
    }
    void *ptr = &g_heap[g_heap_idx];
    g_heap_idx += size;
    return ptr;
}

void free(void *ptr) {
    (void)ptr;
}

void *realloc(void *ptr, unsigned long new_size) {
    if (!ptr) return malloc(new_size);
    void *new_ptr = malloc(new_size);
    if (!new_ptr) return (void*)0;
    uint8_t *src = (uint8_t *)ptr;
    uint8_t *dst = (uint8_t *)new_ptr;
    for (unsigned long i = 0; i < new_size; i++) {
        dst[i] = src[i];
    }
    return new_ptr;
}

void abort(void) {
    while(1);
}

// string.h functions:

unsigned long strlen(const char *s) {
    unsigned long len = 0;
    while (s[len]) len++;
    return len;
}

int custom_strncmp(const uint8_t *s1, const char *s2, unsigned long n) {
    while (n && *s1 && (*s1 == (uint8_t)*s2)) {
        s1++;
        s2++;
        n--;
    }
    if (n == 0) return 0;
    return *s1 - (uint8_t)*s2;
}

// WIT function example:
// __attribute__((export_name("<module_world>#<function_name>")))
// <return type> exports_<module_world>_<function_name>(<parameters>) {}
__attribute__((export_name("ui-world#get-commands")))
void exports_ui_world_get_commands(ui_world_list_command_desc_t *ret) {}
```

## Go

The most painful one to implement...

Create your main file in `libs/<module_name>/export_<module_world>/main.go` and add these:

```go
package export_<module_world> // package name

import (
	// imports generated by wit-bindgen as directories
	"wit_component/local_zappy_command"
	"wit_component/local_zappy_graphic"
	"wit_component/local_zappy_host_api"
	"wit_component/local_zappy_input"
	"wit_component/local_zappy_system"
)
```
